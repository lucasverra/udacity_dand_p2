    # Creates a df that provide with columns = "columns of main_df to analyse"
    #and a row with the quantity of different values that exists in those columns   
    list_of_col = df.columns.values

        np.array([list_of_q_dif_value]),
        columns=list_of_col

def condition_woman_children_passenger(sexe,age):
    if age <= 16 :
        return 'children'
    elif np.isnan(age) :
        return "N/A"
    elif sexe == 'female' :
        return 'woman'
    elif sexe == 'male' :
        return 'not_woman'

#we invoque a fct written in the pandas tutorial
def correlation(x, y):
    '''
    Fill in this function to compute the correlation between the two
    input variables. Each input is either a NumPy array or a Pandas
    Series.
    
    correlation = average of (x in standard units) times (y in standard units)
    
    Remember to pass the argument "ddof=0" to the Pandas std() function!
    '''
    x_norm = (x-x.mean())/x.std(ddof=0)
    y_norm = (y-y.mean())/y.std(ddof=0)
    pearson_r = (x_norm*y_norm).mean()
    return pearson_r

def make_autopct(values):
    def my_autopct(pct):
        total = sum(values)
        val = int(round(pct*total/100.0))
        return '{p:.2f}%  ({v:d})'.format(p=pct,v=val)
    return my_autopct

Limitations:

This work has 4 main limitations:

  1. The representiviness of this sample (891 rows) over the total Titanic population. Fact is we do not how/when was this sample created, so we make comment on the "quality of the sample" and thus, "quality of the analysis"
  2. This analysis is poor on statistical demonstrations. The limit is that the analysis can easyly be challenged so this could represent a vector for improvement for a v2.
  3. A whole part of the data is NOT used, as names (where we could draw familie ties) and study the likelyhood of "family survival"
  4. Missing values in the cabin made impossible the use of that data. It would have been interesting to see if there is correlation between geografical space (cabin XYY) and survival rate.
